  \item Textbook Exercise 6.4.1a
        \begin{solution}
          \textbf{Improved Euler (Midpoint) Method}

          The improved Euler method is:
          \begin{align*}
            v_{i+1/2} & = u_i + \frac{h}{2} f(t_i, u_i)                      \\
            u_{i+1}   & = u_i + h f\left(t_i + \frac{h}{2}, v_{i+1/2}\right)
          \end{align*}

          \medskip

          \textbf{(a) $u' = -2tu$, $0 \leq t \leq 2$, $u(0) = 2$; $h = 0.2$}

          Here $f(t,u) = -2tu$.

          Initial condition: $t_0 = 0$, $u_0 = 2$.

          \textit{Step 1: Compute midpoint value}
          \begin{align*}
            v_{1/2} & = u_0 + \frac{h}{2} f(t_0, u_0)                \\
                    & = 2 + \frac{0.2}{2} \cdot (-2 \cdot 0 \cdot 2) \\
                    & = 2 + 0.1 \cdot 0                              \\
                    & = 2
          \end{align*}

          \textit{Step 2: Compute next value}
          \begin{align*}
            u_1 & = u_0 + h f\left(t_0 + \frac{h}{2}, v_{1/2}\right) \\
                & = 2 + 0.2 \cdot f(0.1, 2)                          \\
                & = 2 + 0.2 \cdot (-2 \cdot 0.1 \cdot 2)             \\
                & = 2 + 0.2 \cdot (-0.4)                             \\
                & = 2 - 0.08                                         \\
                & = 1.92
          \end{align*}

          \textbf{Result:} $\boxed{u_1 = 1.92}$ at $t_1 = 0.2$

          \textit{Comparison with exact solution:}
          \begin{align*}
            \hat{u}(0.2) = 2e^{-(0.2)^2} = 2e^{-0.04} \approx 1.9216
          \end{align*}

          Error: $|u_1 - \hat{u}(0.2)| = |1.92 - 1.9216| = 0.0016$
        \end{solution}
  \item Textbook Exercise 6.4.2a
        \begin{solution}
          \textbf{Modified Euler Method (Forward and Backward Average)}

          The modified Euler method is:
          \begin{align*}
            v_{i+1} & = u_i + h f(t_i, u_i)                                   \\
            u_{i+1} & = u_i + \frac{h}{2} [f(t_i, u_i) + f(t_{i+1}, v_{i+1})]
          \end{align*}

          \medskip

          \textbf{(a) $u' = -2tu$, $0 \leq t \leq 2$, $u(0) = 2$; $h = 0.2$}

          Here $f(t,u) = -2tu$.

          Initial condition: $t_0 = 0$, $u_0 = 2$.

          \textit{Step 1: Compute forward Euler prediction}
          \begin{align*}
            v_1 & = u_0 + h f(t_0, u_0)                \\
                & = 2 + 0.2 \cdot (-2 \cdot 0 \cdot 2) \\
                & = 2 + 0.2 \cdot 0                    \\
                & = 2
          \end{align*}

          \textit{Step 2: Compute average of forward and backward slopes}
          \begin{align*}
            u_1 & = u_0 + \frac{h}{2} [f(t_0, u_0) + f(t_1, v_1)]           \\
                & = 2 + \frac{0.2}{2} [f(0, 2) + f(0.2, 2)]                 \\
                & = 2 + 0.1 [(-2 \cdot 0 \cdot 2) + (-2 \cdot 0.2 \cdot 2)] \\
                & = 2 + 0.1 [0 + (-0.8)]                                    \\
                & = 2 + 0.1 \cdot (-0.8)                                    \\
                & = 2 - 0.08                                                \\
                & = 1.92
          \end{align*}

          \textbf{Result:} $\boxed{u_1 = 1.92}$ at $t_1 = 0.2$

          \textit{Comparison with exact solution:}
          \begin{align*}
            \hat{u}(0.2) = 2e^{-(0.2)^2} = 2e^{-0.04} \approx 1.9216
          \end{align*}

          Error: $|u_1 - \hat{u}(0.2)| = |1.92 - 1.9216| = 0.0016$
        \end{solution}
  \item Textbook Exercise 6.4.6a
        \begin{solution}
          \textbf{Runge-Kutta 4 Method (RK4)}

          The RK4 method is:
          \begin{align*}
            k_1     & = f(t_i, u_i)                                           \\
            k_2     & = f\left(t_i + \frac{h}{2}, u_i + \frac{h}{2}k_1\right) \\
            k_3     & = f\left(t_i + \frac{h}{2}, u_i + \frac{h}{2}k_2\right) \\
            k_4     & = f(t_i + h, u_i + hk_3)                                \\
            u_{i+1} & = u_i + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4)
          \end{align*}

          \medskip

          \textbf{(a) $u' = -2tu$, $0 \leq t \leq 2$, $u(0) = 2$; $h = 0.2$}

          Here $f(t,u) = -2tu$.

          Initial condition: $t_0 = 0$, $u_0 = 2$.

          \textit{Step 1: Compute $k_1$}
          \begin{align*}
            k_1 & = f(0, 2) = -2 \cdot 0 \cdot 2 = 0
          \end{align*}

          \textit{Step 2: Compute $k_2$}
          \begin{align*}
            k_2 & = f\left(0 + \frac{0.2}{2}, 2 + \frac{0.2}{2} \cdot 0\right) \\
                & = f(0.1, 2) = -2 \cdot 0.1 \cdot 2 = -0.4
          \end{align*}

          \textit{Step 3: Compute $k_3$}
          \begin{align*}
            k_3 & = f\left(0.1, 2 + 0.1 \cdot (-0.4)\right)         \\
                & = f(0.1, 1.96) = -2 \cdot 0.1 \cdot 1.96 = -0.392
          \end{align*}

          \textit{Step 4: Compute $k_4$}
          \begin{align*}
            k_4 & = f(0 + 0.2, 2 + 0.2 \cdot (-0.392))                    \\
                & = f(0.2, 1.9216) = -2 \cdot 0.2 \cdot 1.9216 = -0.76864
          \end{align*}

          \textit{Step 5: Compute $u_1$}
          \begin{align*}
            u_1 & = u_0 + \frac{h}{6}(k_1 + 2k_2 + 2k_3 + k_4)              \\
                & = 2 + \frac{0.2}{6}(0 + 2(-0.4) + 2(-0.392) + (-0.76864)) \\
                & = 2 + \frac{0.2}{6}(0 - 0.8 - 0.784 - 0.76864)            \\
                & = 2 + \frac{0.2}{6}(-2.35264)                             \\
                & = 2 + \frac{-0.470528}{6}                                 \\
                & = 2 - 0.0784213...                                        \\
                & \approx 1.9216
          \end{align*}

          \textbf{Result:} $\boxed{u_1 \approx 1.9216}$ at $t_1 = 0.2$

          \textit{Comparison with exact solution:}
          \begin{align*}
            \hat{u}(0.2) = 2e^{-(0.2)^2} = 2e^{-0.04} \approx 1.9216
          \end{align*}

          Error: $|u_1 - \hat{u}(0.2)| \approx 0$ (essentially exact to 4 decimal places!)
        \end{solution}
  \item Textbook Exercise 6.4.11
        \begin{solution}
          Consider the problem $u' = cu$, $u(0) = 1$ for a real constant $c$ and $t > 0$.

          \medskip

          \textbf{(a) Find an explicit formula for $u_{i+1}/u_i$ in the modified Euler method}

          The modified Euler method is:
          \begin{align*}
            v_{i+1} & = u_i + h f(t_i, u_i)                                   \\
            u_{i+1} & = u_i + \frac{h}{2} [f(t_i, u_i) + f(t_{i+1}, v_{i+1})]
          \end{align*}

          For $f(t,u) = cu$:

          \textit{Step 1: Compute predictor}
          \begin{align*}
            v_{i+1} & = u_i + h(cu_i) = u_i(1 + ch)
          \end{align*}

          \textit{Step 2: Compute corrector}
          \begin{align*}
            u_{i+1} & = u_i + \frac{h}{2}[cu_i + cv_{i+1}]            \\
                    & = u_i + \frac{h}{2}[cu_i + c \cdot u_i(1 + ch)] \\
                    & = u_i + \frac{ch}{2}u_i[1 + (1 + ch)]           \\
                    & = u_i + \frac{ch}{2}u_i[2 + ch]                 \\
                    & = u_i\left[1 + \frac{ch}{2}(2 + ch)\right]      \\
                    & = u_i\left[1 + ch + \frac{(ch)^2}{2}\right]
          \end{align*}

          Therefore:
          \begin{align*}
            \boxed{\frac{u_{i+1}}{u_i} = 1 + ch + \frac{(ch)^2}{2}}
          \end{align*}

          \medskip

          \textbf{(b) Show that if $ch = -3$, then $|u_i| \to \infty$ while $\hat{u}(t) \to 0$ as $t \to \infty$}

          \textit{For the numerical solution:}

          When $ch = -3$:
          \begin{align*}
            \frac{u_{i+1}}{u_i} & = 1 + (-3) + \frac{(-3)^2}{2} \\
                                & = 1 - 3 + \frac{9}{2}         \\
                                & = -2 + 4.5                    \\
                                & = 2.5
          \end{align*}

          Since $u_0 = 1$, we have:
          \begin{align*}
            u_i = (2.5)^i
          \end{align*}

          Therefore $|u_i| = (2.5)^i \to \infty$ as $i \to \infty$. \quad $\checkmark$

          \textit{For the exact solution:}

          The exact solution is:
          \begin{align*}
            \hat{u}(t) = e^{ct}
          \end{align*}

          Since $ch = -3$ and $h > 0$, we have $c < 0$ (specifically $c = -3/h < 0$).

          Therefore:
          \begin{align*}
            \hat{u}(t) = e^{ct} \to 0 \quad \text{as } t \to \infty \quad \checkmark
          \end{align*}

          \textbf{Conclusion:} The modified Euler method is unstable for this problem when $ch = -3$, causing the numerical solution to diverge even though the exact solution decays to zero.
        \end{solution}


  \item It is possible to generalize Euler's method using any finite difference approximation to $u''(t)$.
        \begin{parts}
          \part Let $k \in (0, 1)$ be a constant. Construct a FD approximation for $u''(t)$ using $u'(t)$ and $u'(t+kh)$.
          \part Derive a new one-step method for solving $u' = f(t,u)$.
          \part What is the order of accuracy of this method?
          \part Write the Butcher tableau for this method.
        \end{parts}

        \begin{solution}
          \begin{parts}
            \part The Taylor expansion of $u'(t+kh)$ around $t$ is:
            \begin{align*}
              u''(t) & = \frac{u'(t+kh) - u'(t)}{kh} + O(h) \\
            \end{align*}
            \part 
            The Taylor expansion of $u(t+h)$ around $t$ is:
             \begin{align*}
              u(t+h) & = u(t) + h u'(t) + \frac{h^2}{2} u''(t) + O(h^3) \\
                     & = u(t) + h u'(t) + \frac{h^2}{2} \left( \frac{u'(t+kh) - u'(t)}{kh} \right) + O(h^3) \\
                     & = u(t) + h u'(t) + \frac{h}{2k} (u'(t+kh) - u'(t)) + O(h^3) \\
                     & = u(t) + h \left(1 - \frac{1}{2k}\right) u'(t) + \frac{h}{2k} u'(t+kh) + O(h^3) \\
                     & = u(t) + h \left(1 - \frac{1}{2k}\right) f(t, u(t)) + \frac{h}{2k} f(t+kh, u(t+kh)) + O(h^3)
            \end{align*}
            \part 
          This gives us a new one-step method:
          \begin{align*}
            k_1 & = hf(t_i, u_i) \\
            k_2 & = hf(t_i + kh, u_i + k k_1) \\
            u_{i+1} & = u_i + \left(1 - \frac{1}{2k}\right) k_1 + \frac{1}{2k} k_2
          \end{align*}
          \part The local truncation error is $O(h^3)$.
          \part The Butcher tableau for this method is:
          \[
            \begin{array}{c|cc}
              0   &      &     \\[0.5em]
              k   & k    &     \\[0.5em]
              \hline
                  & 1 - \frac{1}{2k} & \frac{1}{2k} \\
            \end{array}
          \]
          \end{parts}
        \end{solution}
  \item The classic Runge-Kutta method (called the 3/8th rule) uses a slightly different Butcher tableau than the one we discussed in class:
        \[
          \begin{array}{c|cccc}
            0   &      &     &           \\[0.5em]
            1/3 & 1/3  &     &           \\[0.5em]
            2/3 & -1/3 & 1   &           \\[0.5em]
            1   & 1    & -1  & 1         \\[0.5em]
            \hline
                & 1/8  & 3/8 & 3/8 & 1/8 \\
          \end{array}
        \]
        Note that any \emph{missing} entries in the Butcher tableau are assumed to be zero.
        \begin{parts}
          \part Explicitly write out the update formula for this method.
          \part What integration method does this correspond to when applied to the integral form of the IVP? (This is in fact another numerical integration method called Simpson's 3/8th rule.)
        \end{parts}
        \begin{solution}
          \begin{parts}
            \part The update formula for this method is:
            \begin{align*}
              k_1 & = hf(t_i, u_i) \\
              k_2 & = hf\left(t_i + \frac{h}{3}, u_i + \frac{1}{3}k_1\right) \\
              k_3 & = hf\left(t_i + \frac{2h}{3}, u_i - \frac{1}{3}k_1 + k_2\right) \\
              k_4 & = hf\left(t_i + h, u_i + k_1 - k_2 + k_3\right) \\
              u_{i+1} & = u_i + \frac{1}{8}k_1 + \frac{3}{8}k_2 + \frac{3}{8}k_3 + \frac{1}{8}k_4
            \end{align*}
            \part When applied to the integral form of the IVP, this corresponds to Simpson's 3/8th rule for numerical integration:
            \begin{align*}
              \int_{t_i}^{t_{i+1}} u'(t) dt & \approx \frac{h}{8}\left[u(t_i) + 3u\left(t_i + \frac{h}{3}\right) + 3u\left(t_i + \frac{2h}{3}\right) + u(t_{i+1})\right]
            \end{align*}
          \end{parts}
        \end{solution}
