\item Textbook Exercise 5.4.7
\begin{solution}
\textbf{(a)} The centered difference formula converges to $f'(x)$ as $h \to 0$.

\textbf{(b)} For convergence of a general finite-difference formula $f'(x) \approx \frac{1}{h}\sum_{k=-p}^{q} a_k f(x+kh)$:

$$\boxed{\sum_{k=-p}^{q} a_k = 0}$$

$$\boxed{\sum_{k=-p}^{q} k a_k = 1}$$
\end{solution}

\item Richardson extrapolation. Let $D(h)$ be a finite difference approximation to $f'(a)$ with truncation error $O(h^p)$. Construct the combined estimate:
\[
R(h) = x D(h) + y D(2h)
\]

\begin{solution}
\textbf{Part 1.} Convergence condition: $\boxed{x + y = 1}$

\textbf{Part 2.} For improved convergence:
\[
\boxed{x = \frac{2^p}{2^p - 1}, \quad y = \frac{-1}{2^p - 1}}
\]
Order of convergence: $\boxed{O(h^{p+1})}$

\textbf{Part 3.} For forward difference ($p = 1$):
\[
\boxed{R(h) = \frac{4f(a+h) - f(a+2h) - 3f(a)}{2h}, \quad \text{Order: } O(h^2)}
\]

\end{solution}
