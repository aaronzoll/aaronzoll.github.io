  \question[10]

  \begin{parts}
    \part Find the derivative of $\displaystyle f(x) = \frac{x^2 e^x}{\cos(x)}$. You don't need to simplify your final answer.

\begin{solution}
  Use the quotient rule: $\displaystyle \frac{d}{dx}\left[\frac{u}{v}\right] = \frac{u'v - uv'}{v^2}$
  
  Let $u = x^2 e^x$ and $v = \cos(x)$.
  
  For $u' = \frac{d}{dx}(x^2 e^x)$, use the product rule:
  $$u' = 2xe^x + x^2 e^x = e^x(2x + x^2)$$
  
  For $v'$:
  $$v' = -\sin(x)$$
  
  Therefore:
  \begin{align*}
    f'(x) &= \frac{e^x(2x + x^2) \cdot \cos(x) - x^2 e^x \cdot (-\sin(x))}{\cos^2(x)} \\
  \end{align*}
  
  \textbf{Answer:} $\displaystyle f'(x) = \frac{x e^x[(2 + x)\cos(x) + x\sin(x)]}{\cos^2(x)}$
\end{solution}
    \vfill

    \part 
    For the curve defined by $x^2 y + \sin(y) = x$, find the equation of the tangent line at the point $(0, 0)$.

    \begin{solution}
      Differentiate implicitly:
      $$2xy + x^2\frac{dy}{dx} + \cos(y)\frac{dy}{dx} = 1$$

      $$\frac{dy}{dx}(x^2 + \cos(y)) = 1 - 2xy$$

      At $(0,0)$: $\displaystyle \frac{dy}{dx}\bigg|_{(0,0)} = \frac{1 - 0}{0 + \cos(0)} = \frac{1}{1} = 1$

      Tangent line: $y - 0 = 1(x - 0)$

      \textbf{Answer:} $y = x$
    \end{solution}

    \vfill
  \end{parts}

  \question[10]

  \begin{parts}
    \part Evaluate $\displaystyle \int xe^{(x^2)} - x\cos(x) \, dx$
    \begin{solution}
      Split into two integrals:

      For $\int xe^{x^2} dx$: Let $u = x^2$, $du = 2x\,dx$, so $\int xe^{x^2}dx = \frac{1}{2}e^{x^2} + C_1$

      For $\int x\cos(x) dx$: Integration by parts with $u = x$, $dv = \cos(x)dx$:
      $$\int x\cos(x)dx = x\sin(x) - \int \sin(x)dx = x\sin(x) + \cos(x) + C_2$$

      \textbf{Answer:} $\displaystyle \frac{1}{2}e^{x^2} - x\sin(x) - \cos(x) + C$
    \end{solution}
    \vfill


    \part Let $f(x) = x^3 - 6x^2 + 9x + 1$ on the interval $[0, 5]$.

    Find the absolute maximum and minimum values of $f$ on this interval.

    \begin{solution}
      Find critical points: $f'(x) = 3x^2 - 12x + 9 = 3(x-1)(x-3) = 0$ gives $x = 1, 3$.

      Evaluate at critical points and endpoints:
      \begin{align*}
        f(0) & = 1                       \\
        f(1) & = 1 - 6 + 9 + 1 = 5       \\
        f(3) & = 27 - 54 + 27 + 1 = 1    \\
        f(5) & = 125 - 150 + 45 + 1 = 21
      \end{align*}

      \textbf{Absolute maximum:} $21$ at $x = 5$

      \textbf{Absolute minimum:} $1$ at $x = 0$ and $x = 3$
    \end{solution}

    \vfill
  \end{parts}
